#### S008 - 重试机制
```python
import time
from functools import wraps

def retry(max_attempts=3, delay=2):
    def decorator(f):
        @wraps(f)
        def wrapper(*args, **kwargs):
            for i in range(max_attempts):
                try:
                    return f(*args, **kwargs)
                except Exception as e:
                    if i == max_attempts - 1:
                        raise
                    time.sleep(delay * (i + 1))
            return None
        return wrapper
    return decorator
```

#### S012 - SQLite 查询缓存
```python
import hashlib

class QueryCache:
    def __init__(self, db_path="query_cache.db"):
        self.conn = sqlite3.connect(db_path)
        self.conn.execute("CREATE TABLE IF NOT EXISTS qcache (h TEXT PRIMARY KEY, r TEXT, t DATETIME)")
    
    def get(self, sql, params):
        h = hashlib.md5((sql + str(params)).encode()).hexdigest()
        row = self.conn.execute("SELECT r FROM qcache WHERE h=? AND t > datetime('now', '-1 hour')", (h,)).fetchone()
        return json.loads(row[0]) if row else None
    
    def set(self, sql, params, result):
        h = hashlib.md5((sql + str(params)).encode()).hexdigest()
        self.conn.execute("INSERT OR REPLACE INTO qcache VALUES (?, ?, datetime('now'))",
                          (h, json.dumps(result, ensure_ascii=False)))
        self.conn.commit()
```

### 2.7 文件操作

#### S015 - 安全文件写入
```python
import tempfile
import os

def safe_write(path, content):
    """先写临时文件再重命名，防止写一半崩溃"""
    fd, tmp = tempfile.mkstemp(dir=os.path.dirname(path))
    try:
        with os.fdopen(fd, 'w') as f:
            f.write(content)
        os.replace(tmp, path)
    except:
        os.unlink(tmp)
        raise
```
