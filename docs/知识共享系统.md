## 2. 知识库结构

### 2.1 数据库表

```sql
-- 知识条目表
CREATE TABLE knowledge_items (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,             -- 标题
    content TEXT NOT NULL,           -- 内容
    category TEXT NOT NULL,          -- 分类: tech/biz/lesson
    tags TEXT,                       -- 标签（逗号分隔）
    author TEXT,                     -- 贡献者
    task_id TEXT,                    -- 关联任务
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    view_count INTEGER DEFAULT 0,    -- 查看次数
    useful_count INTEGER DEFAULT 0,   -- 有用次数
    version INTEGER DEFAULT 1
);

-- 全文索引
CREATE VIRTUAL TABLE knowledge_fts USING fts5(
    title, content, tags,
    content_rowid=id
);
```

## 4. 知识检索

### 4.1 关键词检索

```python
def search_knowledge(keywords: str, category: str = None, limit: int = 10):
    """搜索知识库"""
    conn = sqlite3.connect("/home/admin/xinhai_legal_api/knowledge_base.db")
    cursor = conn.cursor()
    
    if category:
        cursor.execute("""
            SELECT k.* FROM knowledge_items k
            JOIN knowledge_fts fts ON k.id = fts.rowid
            WHERE knowledge_fts MATCH ? AND k.category = ?
            ORDER BY rank
            LIMIT ?
        """, (keywords, category, limit))
    else:
        cursor.execute("""
            SELECT k.* FROM knowledge_items k
            JOIN knowledge_fts fts ON k.id = fts.rowid
            WHERE knowledge_fts MATCH ?
            ORDER BY rank
            LIMIT ?
        """, (keywords, limit))
    
    return [dict(row) for row in cursor.fetchall()]
```

### 4.2 语义检索

```python
def semantic_search(query: str, limit: int = 5):
    """语义搜索"""
    chroma = ChromaSearch()
    results = chroma.search(query, n_results=limit)
    return results
```
